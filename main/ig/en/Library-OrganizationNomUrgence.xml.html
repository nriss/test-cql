using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

codesystem &quot;TRE_R211-ActiviteOperationnelle&quot;: 'https://mos.esante.gouv.fr/NOS/TRE_R211-ActiviteOperationnelle/FHIR/TRE-R211-ActiviteOperationnelle'
codesystem &quot;TRE_R213-ModePriseEnCharge&quot;: 'https://mos.esante.gouv.fr/NOS/TRE_R213-ModePriseEnCharge/FHIR/TRE-R213-ModePriseEnCharge'

code &quot;Urgences polyvalentes hospitalières&quot;: '157' from &quot;TRE_R211-ActiviteOperationnelle&quot;
code &quot;Urgences pédiatriques hospitalières&quot;: '249' from &quot;TRE_R211-ActiviteOperationnelle&quot;
code &quot;Accueil des urgences&quot;: '33' from &quot;TRE_R213-ModePriseEnCharge&quot;

context Organization

// HealthcareService rattachés à l'Organization courante
define &quot;HealthcareServicesLies&quot;:
  [HealthcareService] HS
    where HS.providedBy.reference = 'Organization/' + Organization.id

// Concernée par l'obligation de nommage : au moins un HealthcareService avec une activité
// opérationnelle &quot;urgences&quot; (157 ou 249) ET un mode de prise en charge &quot;accueil des urgences&quot; (33)
define &quot;ConcerneeParObligationDeNommage&quot;:
  exists (
    &quot;HealthcareServicesLies&quot; HS
      where (
        exists (HS.specialty S where FHIRHelpers.ToConcept(S) ~ &quot;Urgences polyvalentes hospitalières&quot;)
        or exists (HS.specialty S where FHIRHelpers.ToConcept(S) ~ &quot;Urgences pédiatriques hospitalières&quot;)
      )
      and exists (HS.characteristic C where FHIRHelpers.ToConcept(C) ~ &quot;Accueil des urgences&quot;)
  )

define &quot;InitialPopulation&quot;:
  &quot;ConcerneeParObligationDeNommage&quot;

define &quot;Denominator&quot;:
  &quot;InitialPopulation&quot;

// Le nom (ou un alias) de l'Organization contient &quot;urgence&quot;
define &quot;NomContientUrgence&quot;:
  (Organization.name is not null and Lower(Organization.name) contains 'urgence')
    or exists (Organization.alias A where Lower(A) contains 'urgence')

// Numérateur : Organization concernées et NON conformes (pas d'&quot;urgence&quot; dans le nom)
define &quot;Numerator&quot;:
  &quot;InitialPopulation&quot; and not &quot;NomContientUrgence&quot;
